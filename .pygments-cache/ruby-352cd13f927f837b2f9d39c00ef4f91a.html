<pre><code>module ActionView
  def will_paginate(collection = nil, options = {})
    options[:renderer] ||= BootstrapLinkRenderer
    super.try :html_safe
  end

  class BootstrapLinkRenderer &lt; LinkRenderer
    protected

    def html_container(html)
      tag :div, tag(:ul, html), container_attributes
    end

    def page_number(page)
      tag :li, link(page, page, :rel =&gt; rel_value(page)), :class =&gt; ('active' if page == current_page)
    end

    def previous_or_next_page(page, text, classname)
      tag :li, link(text, page || '#'), :class =&gt; [classname[0..3], classname, ('disabled' unless page)].join(' ')
    end

    def gap
      tag :li, link(super, '#'), :class =&gt; 'disabled'
    end
  end
end
